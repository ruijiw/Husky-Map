003/** <b>Edge</b> represents an <b> immutable </b> connection between 
004 * two Nodes in Graph. It starts from one node and points
005 * to another node.
006 * It includes a Node that represents the parent Node it 
007 * starts from, a Node that represents the children Node 
008 * it ends with and a String that represents the label of itself. 
009 * <p>
010 * Examples of Edge include "Edge:label<node1, node2>" and so on.
011*/
012
013public class Edge<N extends Comparable<N>, E extends Comparable<E>> implements Comparable<Edge<N, E>>{
014        private final E label;
015        private final Node<N> node1;
016        private final Node<N> node2;
017        
018        // Abstract Function:
019        // the label of an Edge is mapped to label
020        // the parent Node of an Edge is mapped to node1
021        // the children Node of an Edge is mapped to node2
022                
023        // Representation invariant:
024        // data != null &&
025        // node1 != null &&
026        // node2 != null
027        
028        /**
029         * @param label the label of new Edge
030         * @param node1 the parent Node of new Edge
031         * @param node2 the children Node of new Edge
032         * @effects constructs a new Edge with a label,
033         * a parent Node and a children Node
034         */
035        public Edge(E label, Node<N> node1, Node<N> node2) {
036                this.label = label;
037                this.node1 = node1;
038                this.node2 = node2;
039                checkRep();
040        }
041        
042        /** 
043         * check whether the representation invariant holds (if any)
044         */
045        private void checkRep() {
046                assert(label != null) : "label of an Edge cannot be null";
047                assert(node1 != null) : "parent Node of an Edge cannot be null";
048                assert(node2 != null) : "children Node of an Edge cannot be null";
049        }
050
051        /** Compares two Edge.The parent node first, label second 
052         * and children node last
053         * @param edge The Edge to be compared.
054         * @requires the compared edge is not null
055         * @return a negative number if this < edge,
056         * 0 if this = edge,
057         * a positive number if this > edge.
058         */
059        @Override
060        public int compareTo(Edge<N, E> edge) {
061                int compare1 = this.node1.compareTo(edge.node1);
062                int compare2 = this.node2.compareTo(edge.node2);
063                if (compare1 == 0) {
064                        if (compare2 == 0) {
065                                return this.label.compareTo(edge.label);
066                        }
067                        return compare2;
068                }
069                return compare1;
070        }
071        
072        /**
073         * Returns the label of this Edge
074         * @return the label of this Edge
075         */
076        public E getLabel() {
077                return this.label;
078        }
079        
080        /**
081         * Returns the parent Node of this Edge
082         * @return the parent Node of this Edge
083         */
084        public Node<N> getParent() {
085                return this.node1;
086        }
087        
088        /**
089         * Returns the children Node of this Edge
090         * @return the children Node of this Edge
091         */
092        public Node<N> getChildren() {
093                return this.node2;
094        }
095        
096        /** Standard hashCode function.
097         * @return an integer that all objects equal to this will also return.
098         */
099        @Override
100        public int hashCode() {
101                return label.hashCode() + node1.hashCode() + node2.hashCode();
102        }
103        
104        /** Standard equality operation.
105         * @param obj The object to be compared for equality.
106         * @return true if and only if 'obj' is an instance of a Edge
107         * and 'this' and 'obj' represent the same Edge.
108         */
109        @Override
110        public boolean equals(Object obj) {
111                if (obj instanceof Edge<?, ?>) {
112                        Edge<?, ?> edge = (Edge<?, ?>) obj;
113                        if (this.hashCode() != edge.hashCode()) {
114                          return false;
115                    }
116                        return edge.label.equals(this.label) && edge.node1.equals(this.node1)
117                        && edge.node2.equals(this.node2);
118                }
119                return false;
120        }
121        
122        /**
123         * Returns a String that represents this Edge
124         * @return a String representation of this Edge that contains its label,
125         * its parent Node and its children Node. 
126         */
127        public String toString() {
128                return "Edge:" + this.label.toString() + "<" + node1.getData().toString() 
129                                + ", " + node2.getData().toString() + ">";
130        }
131}
