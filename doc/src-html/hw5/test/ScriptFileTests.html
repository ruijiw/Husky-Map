001package hw5.test;
002
003import static org.junit.Assert.assertEquals;
004
005import java.io.*;
006import java.net.URISyntaxException;
007import java.util.*;
008
009import org.junit.Test;
010import org.junit.runner.RunWith;
011import org.junit.runners.Parameterized.Parameters;
012
013import utils.tests.LabelledParameterized;
014import utils.tests.LabelledParameterized.Labels;
015
016/**
017 * This class, along with a complete HW5TestDriver implementation,
018 * can be used to test the your implementations of Graph and the
019 * path finding algorithm using the script file format described
020 * in the problem set.  It is assumed that the files are
021 * located in the same directory as this class.
022 *
023 * It works by parameterizing test methods over some data values, and then
024 * creating an instance for the cross-product of test methods and data values.
025 * In this case, it will create one ScriptFileTests instance per .expected file,
026 * and for each of those it will run the checkAgainstExpectedOutput() test.
027 * See the JUnit4 Javadocs for more information, or Google for more examples.
028 */
029@RunWith(LabelledParameterized.class)
030public class ScriptFileTests {
031
032  //static fields and methods used during setup of the parameterized runner
033  private static FileFilter testFileFilter = new FileFilter() {
034    @Override
035    public boolean accept(File file) {
036      return file.getName().endsWith(".test");
037    }
038  };
039  private static List<String> testScriptNames = null; // not yet calculated
040  private static List<File> testScriptFiles = null; // not yet calculated
041
042  //used by the actual test instance
043  private final File testScriptFile;
044
045  /**
046   * This method searches for and creates file handles for each script test.
047   * It only searches the immediate directory where the ScriptFileTests.class
048   * classfile is located.
049   */
050  public static void calculateTestFiles() {
051    if ((ScriptFileTests.testScriptFiles != null)
052        || (ScriptFileTests.testScriptNames != null)) {
053      //already initialized
054      return;
055    }
056
057    ScriptFileTests.testScriptNames = new LinkedList<String>();
058    ScriptFileTests.testScriptFiles = new LinkedList<File>();
059    try {
060      // getResource() cannot be null: this file itself is ScriptFileTests
061      // getParentFile() cannot be null: ScriptFileTests has a package
062      @SuppressWarnings("nullness")
063      File myDirectory = new File(ScriptFileTests.class.getResource("ScriptFileTests.class").toURI()).getParentFile();
064      for (File f : myDirectory.listFiles(ScriptFileTests.testFileFilter)) {
065        testScriptNames.add(f.getName());
066        testScriptFiles.add(f);
067      }
068
069    } catch (URISyntaxException e) {
070      throw new RuntimeException(e);
071    }
072  }
073
074  /**
075   * This method is called in the constructor of Parameterized.
076   *
077   * @return List of argument arrays that should be invoked on the ScriptFileTests constructor by the
078   * Parameterized test runner. Since that runner's constructor has one parameter, the
079   * array only has one element.
080   */
081  @Parameters
082  public static List<Object[]> getTestFiles() {
083    ScriptFileTests.calculateTestFiles();
084
085    if (ScriptFileTests.testScriptFiles == null) {
086      throw new IllegalStateException("Did not initialise any files to test");
087    }
088
089    //we have to wrap testScriptFiles here so Parameterized.class receives a list of arg array.
090    List<Object[]> filesToTest = new ArrayList<Object[]>(testScriptFiles.size());
091    for (File f : ScriptFileTests.testScriptFiles) {
092      filesToTest.add(new Object[]{ f });
093    }
094
095    return filesToTest;
096  }
097
098  /**
099   * This method is called in the constructor of LabelledParameterized. Since
100   * getTestFiles (and thus calculateTestFiles()) should have already been
101   * called by the Parameterized constructor, the test script names should already have been computed.
102   *
103   * @return List of labels to be used as names for each of the parameterized tests. These names
104   * are the same as the script file used to run the test.
105   */
106  @Labels
107  public static List<String> getTestLabels() {
108    if (ScriptFileTests.testScriptNames == null) {
109      throw new IllegalStateException("Must initialize list of test names before creating tests.");
110    }
111
112    return ScriptFileTests.testScriptNames;
113  }
114
115  /**
116   * This constructor is reflectively called by the Parameterized runner. It creates
117   * a script file test instance, representing one script file to be tested.
118   */
119  public ScriptFileTests(File testScriptFile) {
120    this.testScriptFile = testScriptFile;
121  }
122
123  /**
124   * Reads in the contents of a file
125   * @throws FileNotFoundException, IOException
126   * @requires that the specified File exists && File ends with a newline
127   * @returns the contents of that file
128   */
129  private String fileContents(File f) throws IOException {
130    if (f == null) {
131      throw new IllegalArgumentException("No file specified");
132    }
133
134    BufferedReader br = new BufferedReader(new FileReader(f));
135
136    StringBuilder result = new StringBuilder();
137    String line = null;
138
139    //read line reads up to *any* newline character
140    while ( (line = br.readLine()) != null) {
141      result.append(line.trim());
142      result.append('\n');
143    }
144
145    br.close();
146    return result.toString();
147  }
148
149  /**
150   * @throws IOException
151   * @requires there exists a test file indicated by testScriptFile
152   *
153   * @effects runs the test in filename, and output its results to a file in
154   * the same directory with name filename+".actual"; if that file already
155   * exists, it will be overwritten.
156   * @returns the contents of the output file
157   */
158  private String runScriptFile() throws IOException {
159    if (testScriptFile == null) {
160      throw new RuntimeException("No file specified");
161    }
162
163    File actual = fileWithSuffix("actual");
164
165    Reader r = new FileReader(testScriptFile);
166    Writer w = new FileWriter(actual);
167
168    HW5TestDriver td = new HW5TestDriver(r, w);
169    td.runTests();
170
171    return fileContents(actual);
172  }
173
174  /**
175   * @param newSuffix
176   * @return a File with the same name as testScriptFile, except that the test
177   *         suffix is replaced by the given suffix
178   */
179  private File fileWithSuffix(String newSuffix) {
180    File parent = testScriptFile.getParentFile();
181    String driverName = testScriptFile.getName();
182    String baseName = driverName.substring(0, driverName.length() - "test".length());
183
184    return new File(parent, baseName + newSuffix);
185  }
186
187  /**
188   * The only test that is run: run a script file and test its output.
189   * @throws IOException
190   */
191  @Test
192  public void checkAgainstExpectedOutput() throws IOException {
193    File expected = fileWithSuffix("expected");
194    assertEquals(testScriptFile.getName(), fileContents(expected), runScriptFile());
195  }
196}
