001package hw5.test;
002
003import static org.junit.Assert.*;
004import hw5.Node;
005
006import org.junit.Test;
007
008/**
009 * This class contains a set of test cases that can be used to test the
010 * implementation of the Node class.
011 */
012
013public class NodeTest {
014        
015        // simple base Nodes
016        private Node<String> integer = new Node<String>("" + 1);
017        private Node<String> empty = new Node<String>("");
018        private Node<String> longNode = new Node<String>("11111111111111111111111111111111111111111111");
019        private Node<String> string = new Node<String>("string");
020        private Node<String> space = new Node<String>(" ");
021        private Node<String> doubleNode = new Node<String>("" + 1.11);
022        
023        // test whether a String representation of a Node is equal to a specified String
024        private void eq(Node<String> node, String rep) {
025            assertEquals(rep, node.toString());
026        }
027
028        ///////////////////////////////////////////////////////////////////////////////////////
029        ////  Constructor
030        /////////////////////////////////////////////////////////////////////////////////////// 
031        
032        @Test
033        public void testConstructor() {
034            new Node<String>("");
035            new Node<String>("11111111111111111111111111111111111111111111");
036            new Node<String>("string");
037            new Node<String>("" + 1);
038            new Node<String>(" ");
039            new Node<String>("1.11");
040        }
041        
042        ///////////////////////////////////////////////////////////////////////////////////////
043        ////  getData
044        /////////////////////////////////////////////////////////////////////////////////////// 
045        
046        @Test
047        public void testgetData() {
048                assertEquals(integer.getData(), "1");
049                assertEquals(empty.getData(), "");
050                assertEquals(longNode.getData(), "11111111111111111111111111111111111111111111");
051                assertEquals(string.getData(), "string");
052                assertEquals(space.getData(), " ");
053                assertEquals(doubleNode.getData(), "1.11");
054        }
055        
056        ///////////////////////////////////////////////////////////////////////////////////////
057        ////  String
058        /////////////////////////////////////////////////////////////////////////////////////// 
059        
060        @Test
061        public void testToString() {
062                eq(integer, "Node: 1");
063                eq(empty, "Node: ");
064            eq(longNode, "Node: " + "11111111111111111111111111111111111111111111");
065            eq(string, "Node: string");
066            eq(space, "Node:  ");
067            eq(doubleNode, "Node: 1.11");
068
069        }
070        
071        ///////////////////////////////////////////////////////////////////////////////////////
072        ////  equals
073        /////////////////////////////////////////////////////////////////////////////////////// 
074        
075        // test the reflexive property of equality
076        @Test
077        public void testEqualsReflexive() {
078                assertEquals(integer, integer);
079                assertEquals(empty, empty);
080                assertEquals(longNode, longNode);
081                assertEquals(string, string);
082                assertEquals(space, space);
083                assertEquals(doubleNode, doubleNode);
084        }
085        
086        // test the symmetric property of equality
087        @Test
088        public void testEqualsSymmetric() {
089                Node<String> duplicateNode = new Node<String>(integer.getData());
090                assertEquals(integer, duplicateNode);
091                assertEquals(duplicateNode, integer);
092                
093                Node<String> duplicateNode1 = new Node<String>(string.getData());
094                assertEquals(string, duplicateNode1);
095                assertEquals(duplicateNode1, string);
096                
097                Node<String> duplicateNode2 = new Node<String>(space.getData());
098                assertEquals(space, duplicateNode2);
099                assertEquals(duplicateNode2, space);
100                
101                Node<String> duplicateNode3 = new Node<String>(longNode.getData());
102                assertEquals(longNode, duplicateNode3);
103                assertEquals(duplicateNode3, longNode);
104                
105                Node<String> duplicateNode4 = new Node<String>(empty.getData());
106                assertEquals(empty, duplicateNode4);
107                assertEquals(duplicateNode4, empty);
108                
109                Node<String> duplicateNode5 = new Node<String>(doubleNode.getData());
110                assertEquals(doubleNode, duplicateNode5);
111                assertEquals(duplicateNode5, doubleNode);
112        }
113        
114        // test the transitive property of equality
115        @Test
116        public void testEqualsTransitive() {
117                Node<String> duplicateNode = new Node<String>(empty.getData());
118                Node<String> copyDuplicate = new Node<String>(duplicateNode.getData());
119                assertEquals(empty, duplicateNode);
120                assertEquals(duplicateNode, copyDuplicate);
121                assertEquals(copyDuplicate, empty);     
122                
123                Node<String> duplicateNode1 = new Node<String>(integer.getData());
124                Node<String> copyDuplicate1 = new Node<String>(duplicateNode1.getData());
125                assertEquals(integer, duplicateNode1);
126                assertEquals(duplicateNode1, copyDuplicate1);
127                assertEquals(copyDuplicate1, integer);  
128                
129                Node<String> duplicateNode2 = new Node<String>(space.getData());
130                Node<String> copyDuplicate2 = new Node<String>(duplicateNode2.getData());
131                assertEquals(space, duplicateNode2);
132                assertEquals(duplicateNode2, copyDuplicate2);
133                assertEquals(copyDuplicate2, space);
134                
135                Node<String> duplicateNode3 = new Node<String>(longNode.getData());
136                Node<String> copyDuplicate3 = new Node<String>(duplicateNode3.getData());
137                assertEquals(longNode, duplicateNode3);
138                assertEquals(duplicateNode3, copyDuplicate3);
139                assertEquals(copyDuplicate3, longNode); 
140                
141                Node<String> duplicateNode4 = new Node<String>(string.getData());
142                Node<String> copyDuplicate4 = new Node<String>(duplicateNode4.getData());
143                assertEquals(string, duplicateNode4);
144                assertEquals(duplicateNode4, copyDuplicate4);
145                assertEquals(copyDuplicate4, string);
146                
147                Node<String> duplicateNode5 = new Node<String>(doubleNode.getData());
148                Node<String> copyDuplicate5 = new Node<String>(duplicateNode5.getData());
149                assertEquals(doubleNode, duplicateNode5);
150                assertEquals(duplicateNode5, copyDuplicate5);
151                assertEquals(copyDuplicate5, doubleNode);
152        }

154        ///////////////////////////////////////////////////////////////////////////////////////
155        ////  hashCode
156        /////////////////////////////////////////////////////////////////////////////////////// 
157        
158        @Test
159        public void testHashCodeSelfConsistent() {
160                assertEquals(integer.hashCode(), integer.hashCode());
161                assertEquals(string.hashCode(), string.hashCode());
162                assertEquals(space.hashCode(), space.hashCode());
163                assertEquals(longNode.hashCode(), longNode.hashCode());
164                assertEquals(empty.hashCode(), empty.hashCode());
165                assertEquals(doubleNode.hashCode(), doubleNode.hashCode());
166        }
167        
168        @Test
169        public void testHashCodeConsistentWithEquality() {
170                Node<String> duplicateNode = new Node<String>(integer.getData());
171                assertEquals(integer.equals(duplicateNode), integer.hashCode() == duplicateNode.hashCode());
172                
173                Node<String> duplicateNode1 = new Node<String>(string.getData());
174                assertEquals(string.equals(duplicateNode1), string.hashCode() == duplicateNode1.hashCode());
175                
176                Node<String> duplicateNode2 = new Node<String>(space.getData());
177                assertEquals(space.equals(duplicateNode2), space.hashCode() == duplicateNode2.hashCode());
178                
179                Node<String> duplicateNode3 = new Node<String>(longNode.getData());
180                assertEquals(longNode.equals(duplicateNode3), longNode.hashCode() == duplicateNode3.hashCode());
181                
182                Node<String> duplicateNode4 = new Node<String>(empty.getData());
183                assertEquals(empty.equals(duplicateNode4), empty.hashCode() == duplicateNode4.hashCode());
184                
185                Node<String> duplicateNode5 = new Node<String>(doubleNode.getData());
186                assertEquals(doubleNode.equals(duplicateNode5), doubleNode.hashCode() == duplicateNode5.hashCode());
187        }

192        
193        @Test
194        public void testCompareTo() {
195                assertEquals(integer.compareTo(string), integer.getData().compareTo(string.getData()));
196                assertEquals(space.compareTo(string), space.getData().compareTo(string.getData()));
197                assertEquals(space.compareTo(longNode), space.getData().compareTo(longNode.getData()));
198                assertEquals(empty.compareTo(longNode), empty.getData().compareTo(longNode.getData()));
199                assertEquals(empty.compareTo(doubleNode), empty.getData().compareTo(doubleNode.getData()));
200                assertEquals(integer.compareTo(doubleNode), integer.getData().compareTo(doubleNode.getData()));
201                assertEquals(string.compareTo(string), string.getData().compareTo(string.getData()));
202        }
