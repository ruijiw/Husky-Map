108                // test the symmetric property of equality

111                assertEquals(empty, duplicateEdge);
112                
113                Edge<String, String> duplicateEdge1 = new Edge<String, String>(integer.getLabel(), integer.getParent(), integer.getChildren());
114                assertEquals(duplicateEdge1, integer);
115                assertEquals(integer, duplicateEdge1);
116                
117                Edge<String, String> duplicateEdge2 = new Edge<String, String>(longEdge.getLabel(), longEdge.getParent(), longEdge.getChildren());
118                assertEquals(duplicateEdge2, longEdge);
119                assertEquals(longEdge, duplicateEdge2);
120                
121                Edge<String, String> duplicateEdge3 = new Edge<String, String>(string.getLabel(), string.getParent(), string.getChildren());
122                assertEquals(duplicateEdge3, string);
123                assertEquals(string, duplicateEdge3);
124                
125                Edge<String, String> duplicateEdge4 = new Edge<String, String>(space.getLabel(), space.getParent(), space.getChildren());
126                assertEquals(duplicateEdge4, space);
127                assertEquals(space, duplicateEdge4);
128                
129                Edge<String, String> duplicateEdge5 = new Edge<String, String>(doubleEdge.getLabel(), doubleEdge.getParent(), doubleEdge.getChildren());

134        @Test
135        public void testEqualsTransitive() {
136                // test the transitive property of equality
137                Edge<String, String> duplicateEdge = new Edge<String, String>(empty.getLabel(), empty.getParent(), empty.getChildren());
138                Edge<String, String> copyDuplicate = new Edge<String, String>(duplicateEdge.getLabel(), duplicateEdge.getParent(), duplicateEdge.getChildren());
139                assertEquals(duplicateEdge, empty);
140                assertEquals(duplicateEdge, copyDuplicate);
141                assertEquals(empty, copyDuplicate);
142                
143                Edge<String, String> duplicateEdge1 = new Edge<String, String>(integer.getLabel(), integer.getParent(), integer.getChildren());
144                Edge<String, String> copyDuplicate1 = new Edge<String, String>(duplicateEdge1.getLabel(), duplicateEdge1.getParent(), duplicateEdge1.getChildren());
145                assertEquals(duplicateEdge1, integer);
146                assertEquals(duplicateEdge1, copyDuplicate1);
147                assertEquals(integer, copyDuplicate1);
148                
149                Edge<String, String> duplicateEdge2 = new Edge<String, String>(longEdge.getLabel(), longEdge.getParent(), longEdge.getChildren());
150                Edge<String, String> copyDuplicate2 = new Edge<String, String>(duplicateEdge2.getLabel(), duplicateEdge2.getParent(), duplicateEdge2.getChildren());
151                assertEquals(duplicateEdge2, longEdge);
152                assertEquals(duplicateEdge2, copyDuplicate2);
153                assertEquals(longEdge, copyDuplicate2);
154                
155                Edge<String, String> duplicateEdge3 = new Edge<String, String>(string.getLabel(), string.getParent(), string.getChildren());
156                Edge<String, String> copyDuplicate3 = new Edge<String, String>(duplicateEdge3.getLabel(), duplicateEdge3.getParent(), duplicateEdge3.getChildren());
157                assertEquals(duplicateEdge3, string);
158                assertEquals(duplicateEdge3, copyDuplicate3);
159                assertEquals(string, copyDuplicate3);   
160                
161                Edge<String, String> duplicateEdge4 = new Edge<String, String>(space.getLabel(), space.getParent(), space.getChildren());
162                Edge<String, String> copyDuplicate4 = new Edge<String, String>(duplicateEdge4.getLabel(), duplicateEdge4.getParent(), duplicateEdge4.getChildren());
163                assertEquals(duplicateEdge4, space);
164                assertEquals(duplicateEdge4, copyDuplicate4);
165                assertEquals(space, copyDuplicate4);
166                
167                Edge<String, String> duplicateEdge5 = new Edge<String, String>(doubleEdge.getLabel(), doubleEdge.getParent(), doubleEdge.getChildren());
168                Edge<String, String> copyDuplicate5 = new Edge<String, String>(duplicateEdge5.getLabel(), duplicateEdge5.getParent(), duplicateEdge5.getChildren());
169                assertEquals(duplicateEdge5, doubleEdge);
170                assertEquals(duplicateEdge5, copyDuplicate5);
171                assertEquals(doubleEdge, copyDuplicate5);
172        }

199        @Test
200        public void testHashCodeConsistentWithEquality() {
201                Edge<String, String> duplicateEdge = new Edge<String, String>(empty.getLabel(), empty.getParent(), empty.getChildren());
202                assertEquals(empty.equals(duplicateEdge), empty.hashCode() == duplicateEdge.hashCode());
203                
204                Edge<String, String> duplicateEdge1 = new Edge<String, String>(integer.getLabel(), integer.getParent(), integer.getChildren());
205                assertEquals(integer.equals(duplicateEdge1), integer.hashCode() == duplicateEdge1.hashCode());
206                
207                Edge<String, String> duplicateEdge2 = new Edge<String, String>(longEdge.getLabel(), longEdge.getParent(), longEdge.getChildren());
208                assertEquals(longEdge.equals(duplicateEdge2), longEdge.hashCode() == duplicateEdge2.hashCode());
209                
210                Edge<String, String> duplicateEdge3 = new Edge<String, String>(string.getLabel(), string.getParent(), string.getChildren());
211                assertEquals(string.equals(duplicateEdge3), string.hashCode() == duplicateEdge3.hashCode());
212                
213                Edge<String, String> duplicateEdge4 = new Edge<String, String>(space.getLabel(), space.getParent(), space.getChildren());
214                assertEquals(space.equals(duplicateEdge4), space.hashCode() == duplicateEdge4.hashCode());
215
