002
003/** <b>Node</b> represents an <b> immutable </b> vertex in a Graph. 
004 * It includes a String that represents the data of itself. 
005 * <p>
006 * Examples of Node include "Node: abc", "Node: def" and so on.
007 */
008
009public class Node<N extends Comparable<N>> implements Comparable<Node<N>>{
010        private final N data;
011        
012        //Abstract Function:
013        // the data of an Node is mapped to data
014
015        // Representation invariant:
016        // data != null 
017        
018        /**
019         * @param data the data of new Node
020         * @effects constructs a Node with the specified data
021         */
022        public Node(N data){
023                this.data = data;
024                checkRep();
025        }
026        
027        /**
028         * Checks that the representation invariant holds (if any).
029         */
030        private void checkRep() {
031                assert(this.data != null): "Data of a Node cannot be null";
032        }
033        
034        /** Compares two Node.
035         * @param node The Node to be compared.
036         * @requires the compared node is not null
037         * @return a negative number if this < node,
038         * 0 if this = node,
039         * a positive number if this > node.
040         */
041        @Override
042        public int compareTo(Node<N> node) {
043                return (this.data).compareTo(node.data);
044        }
045                
046        /**
047         * Returns the data of this Node
048         * @return the data of this Node
049         */
050        public N getData() {
051                return this.data;
052        }
053        
054        /** Standard hashCode function.
055     * @return an int that all objects equal to this will also
056     * return.
057     */
058        @Override
059        public int hashCode() {
060                return data.hashCode();
061        }
062        
063        /**
064         * Standard equality operation.
065         * @param obj The object to be compared for equality.
066         * @return true iff 'obj' is an instance of a Node and 'this' and 'obj'
067         *         represent the same Node. 
068         */
069        @Override
070        public boolean equals(Object obj) {
071                if (obj instanceof Node<?>) {
072                        Node<?> node = (Node<?>) obj;
073                        if (this.hashCode() != node.hashCode()) {
074                                return false;
075                        }
076                        return this.data.equals(node.data);
077                }
078                return false;
079        }
080        
081        /** 
082         * returns a string to represent this node.
083         * @return a String representing the data of this node.
084         */
085        @Override
086        public String toString() {
087                return "Node: " + this.data.toString();
088        }
089}
