004package hw3;
005
006import java.util.*;
007import java.lang.Iterable;
008
009/**
010 * This is a container can be used to contain Balls. The key
011 * difference between a BallContainer and a Box is that a Box has a
012 * finite volume. Once a box is full, a client cannot put in more Balls.
013 */
014public class Box implements Iterable<Ball>, Comparator<Ball> {
015
016    /**
017     * ballContainer is used to internally store balls for this Box
018     */
019    private BallContainer ballContainer;
020    private double max;
021
022    /**
023     * Constructor that creates a new box.
024     * @param maxVolume Total volume of balls that this box can contain.
025     */
026    public Box(double maxVolume) {
027        ballContainer = new BallContainer();
028        max = maxVolume;
029    }
030
031    /**
032     * Implements the Iterable interface for this box.
033     * @return an Iterator over the Ball objects contained
034     * in this box.
035     */
036    public Iterator<Ball> iterator() {
037        return ballContainer.iterator();
038    }
039
040
041    /**
042     * This method is used to add Ball objects to this box of
043     * finite volume.  The method returns true if a ball is
044     * successfully added to the box, i.e., ball is not already in the
045     * box and if the box is not already full; and it returns false,
046     * if ball is already in the box or if the box is too full to
047     * contain the new ball.
048     * @param b Ball to be added.
049     * @requires b != null.
050     * @return true if ball was successfully added to the box,
051     * i.e. ball is not already in the box and if the box is not
052     * already full. Returns false, if ball is already in the box or
053     * if the box is too full to contain the new ball.
054     */
055    public boolean add(Ball b) {
056        if (ballContainer.getVolume() + b.getVolume() > max) {
057                return false;
058        } else {
059                return ballContainer.add(b);
060        }
061    }
062    
063    /**
064     * The method compares the volumes of Ball objects for order.
065     * @param a Ball and b Ball to be compared
066     * @return a negative integer, zero, or a positive integer as the 
067     * first the volume of the first ball is less than, equal to, or 
068     * greater than the second.
069     */
070    public int compare(Ball a, Ball b) {
071                return (int) (a.getVolume() - b.getVolume());
072    }
073
074    /**
075     * This method returns an iterator that returns all the balls in
076     * this box in ascending size, i.e., return the smallest Ball
077     * first, followed by Balls of increasing size.
078     * @return an iterator that returns all the balls in this box in
079     * ascending size.
080     */
081    public Iterator<Ball> getBallsFromSmallest() {
082        Iterator<Ball> itr = this.iterator();
083        List<Ball> list = new ArrayList<Ball>();
084        while (itr.hasNext()) {
085           list.add(itr.next());
086        }
087        Collections.sort(list, this);
088        return list.iterator();
089    }
090
091    /**
092     * Removes a ball from the box. This method returns
093     * <tt>true</tt> if ball was successfully removed from the
094     * container, i.e. ball is actually in the box. You cannot
095     * remove a Ball if it is not already in the box and so ths
096     * method will return <tt>false</tt>, otherwise.
097     * @param b Ball to be removed.
098     * @requires b != null.
099     * @return true if ball was successfully removed from the box,
100     * i.e. ball is actually in the box. Returns false, if ball is not
101     * in the box.
102     */
103    public boolean remove(Ball b) {
104        return ballContainer.remove(b);
105    }
106
107    /**
108     * Each Ball has a volume. This method returns the total volume of
109     * all the Balls in the box.
110     * @return the volume of the contents of the box.
111     */
112    public double getVolume() {
113       return ballContainer.getVolume();
114    }
115
116    /**
117     * Returns the number of Balls in this box.
118     * @return the number of Balls in this box.
119     */
120    public int size() {
121        return ballContainer.size();
122    }
123
124    /**
125     * Empties the box, i.e. removes all its contents.
126     */
127    public void clear() {
128        ballContainer.clear();
129    }
130
131    /**
132     * This method returns <tt>true</tt> if this box contains
133     * the specified Ball. It will return <tt>false</tt> otherwise.
134     * @param b Ball to be checked if its in box
135     * @requires b != null.
136     * @return true if this box contains the specified Ball. Returns
137     * false, otherwise.
138     */
139    public boolean contains(Ball b) {
140        return ballContainer.contains(b);
141    }
142
143}
