004package hw3.test;
005import static org.junit.Assert.assertEquals;
006import hw3.HelloWorld;
007import hw3.HolaWorld;
008import java.io.ByteArrayOutputStream;
009import java.io.FileNotFoundException;
010import java.io.PrintStream;
011import org.junit.Test;
012
013/**
014 * HolaWorldTest is a simple test of the HolaWorld class that you
015 * are to fix.  This test just makes sure that the program
016 * does not crash and that the correct greeting is printed.
017 */
018public class HolaWorldTest {
019
020    /**
021     * Tests that HolaWorld does not crash
022     */
023    @Test
024    public void testCrash() {
025        /* Any exception thrown will be caught by JUnit. */
026        HolaWorld.main(new String[0]);
027    }
028
029    /**
030     * Tests that the HolaWorld greeting is correct.
031     */
032    @Test
033    public void testGreeting() {
034        HolaWorld world = new HolaWorld();
035        assertEquals(HolaWorld.SPANISH_GREETING, world.getGreeting());
036    }
037
038    /**
039     * Tests that the output of HolaWorld.main() is correct.
040     * @throws FileNotFoundException 
041     */
042    @Test
043    public void testMainOutput() throws FileNotFoundException {
044        
045        // Needed for windows line endings
046        System.setProperty("line.separator", "\n");
047        
048        // Redirect System.out to an OutputStream
049        PrintStream sysoutStream = System.out;
050        ByteArrayOutputStream mainOutput = new ByteArrayOutputStream();
051        System.setOut(new PrintStream(mainOutput));
052        
053        HolaWorld.main(new String[0]);
054        System.setOut(sysoutStream);
055
056        assertEquals(HelloWorld.GREETING + "\n" + HolaWorld.SPANISH_GREETING + "\n",
057                    mainOutput.toString());
058    }
059}
